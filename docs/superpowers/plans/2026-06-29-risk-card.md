  ```javascript
  async function loadRiskCard(inn) {
      if (!inn) return;
      const block = document.getElementById('riskCardBlock');
      if (!block) return;
      try {
          const res = await apiFetch(`/risk/${encodeURIComponent(inn)}`);
          if (!res.ok) return;
          const data = await res.json();
          block.style.display = '';
          block.innerHTML = renderRiskCard(data);
      } catch { /* молча пропустить */ }
  }

  function renderRiskCard(data) {
      const status = data.status || '';
      const isActive = /действу/i.test(status);
      const isLiquidating = /ликвидац/i.test(status);
      const color = isActive ? 'var(--success, #22c55e)' : isLiquidating ? 'var(--danger, #ef4444)' : 'var(--warning, #f59e0b)';
      const icon = isActive ? '✓' : isLiquidating ? '✗' : '⚠';

      const flags = Array.isArray(data.riskFlags) && data.riskFlags.length
          ? `<ul style="margin:8px 0 0;padding-left:18px;color:var(--text-secondary);font-size:13px;">
               ${data.riskFlags.map(f => `<li>${escapeHtml(f)}</li>`).join('')}
             </ul>`
          : '';

      return `
          <div style="border:1px solid var(--border);border-radius:12px;padding:16px 20px;background:var(--card-bg);">
              <h4 style="margin:0 0 12px;font-size:15px;color:var(--text-primary);">Проверка надёжности (ЕГРЮЛ)</h4>
              <div style="display:flex;align-items:center;gap:8px;margin-bottom:8px;">
                  <span style="width:24px;height:24px;border-radius:50%;background:${color};color:#fff;display:flex;align-items:center;justify-content:center;font-size:13px;font-weight:700;flex-shrink:0;">${icon}</span>
                  <span style="font-weight:600;color:var(--text-primary);font-size:14px;">${escapeHtml(status)}</span>
              </div>
              ${data.registrationDate ? `<p style="margin:4px 0;font-size:13px;color:var(--text-secondary);">Дата регистрации: ${escapeHtml(data.registrationDate)}</p>` : ''}
              ${data.capital ? `<p style="margin:4px 0;font-size:13px;color:var(--text-secondary);">Уставной капитал: ${escapeHtml(data.capital)}</p>` : ''}
              ${flags}
          </div>
      `;
  }
  ```

- [ ] **Step 2: Добавить функцию showRiskModalBeforeAccept в page-specific script**

  ```javascript
  async function showRiskModalBeforeAccept(proposalId, supplierCompany, onConfirm) {
      const modal = document.getElementById('riskModal');
      const body = document.getElementById('riskModalBody');
      const confirmBtn = document.getElementById('riskModalConfirm');

      modal.style.display = 'flex';
      body.innerHTML = '<p style="color:var(--text-secondary);">Загрузка данных...</p>';
      confirmBtn.onclick = () => { modal.style.display = 'none'; onConfirm(); };

      try {
          // Получить ИНН поставщика из каталога
          const compRes = await apiFetch(`/companies?search=${encodeURIComponent(supplierCompany)}`);
          const companies = compRes.ok ? await compRes.json() : [];
          const supplier = companies.find(c => c.company === supplierCompany);
          const inn = supplier?.inn;

          if (!inn) {
              body.innerHTML = '<p style="color:var(--text-secondary);">ИНН поставщика не указан — проверка недоступна.</p>';
              return;
          }

          const riskRes = await apiFetch(`/risk/${encodeURIComponent(inn)}`);
          if (!riskRes.ok) {
              body.innerHTML = '<p style="color:var(--text-secondary);">Сервис проверки временно недоступен.</p>';
              return;
          }
          const data = await riskRes.json();
          const status = data.status || 'Неизвестно';
          const isActive = /действу/i.test(status);
          const isLiquidating = /ликвидац/i.test(status);
          const color = isActive ? '#22c55e' : isLiquidating ? '#ef4444' : '#f59e0b';
          const flags = Array.isArray(data.riskFlags) && data.riskFlags.length
              ? `<p style="margin:8px 0 0;color:#ef4444;font-size:13px;">⚠ ${data.riskFlags.join('; ')}</p>`
              : '';
          body.innerHTML = `
              <div style="display:flex;align-items:center;gap:10px;">
                  <span style="width:12px;height:12px;border-radius:50%;background:${color};flex-shrink:0;display:inline-block;"></span>
                  <span style="font-size:14px;color:var(--text-primary);font-weight:600;">${escapeHtml(status)}</span>
              </div>
              ${data.registrationDate ? `<p style="margin:8px 0 0;color:var(--text-secondary);font-size:13px);">Зарегистрирована: ${escapeHtml(data.registrationDate)}</p>` : ''}
              ${flags}
          `;
      } catch {
          body.innerHTML = '<p style="color:var(--text-secondary);">Не удалось загрузить данные проверки.</p>';
      }
  }
  ```
